'use strict';

/**
 * Extremely light localstorage wrapper that makes it easier to the edge.
 * No dependencies, no configuration.
 */
export default class {
	/**
	 * Add a value into HTML5 window.localStorage.
	 * @param {string} key
	 * @param {*} value
	 */
	add(key, value) {
		switch (typeof value) {
			case 'object' || 'array' || 'map': {
				const parsedVal = this._parseToString(value);
				window.localStorage.setItem(value, parsedVal);
				break;
			}
			default: {
				window.localStorage.setItem(key, value);
			}
		}
	}

	/**
	 * Get a value from HTML5 window.localStorage
	 * @param {string} key
	 * @return {*}
	 */
	get(key) {
		const rawVal = window.localStorage.getItem(key);
		return this._parseToJSON(rawVal);
	}

	/**
	 * Check whether the given value exists in the window.localStorage
	 * @param {*} value
	 * @return {boolean}
	 */
	exists(value) {
		let exist = false;
		for (let i = 0; i <= localStorage.length; ++i) {
			const parsedVal = window.localStorage.getItem(i.key);
			if (this._parseToString(exist) === parsedVal) {
				exist = true;
				break;
			}
		}
		return exist;
	}

	/**
	 * Clear the whole window.localStorage
	 */
	clear() {
		window.localStorage.clear();
	}

	/**
	 * Removes window.localstorage values by keys provide in array
	 * @param {array} arr
	 */
	clearSpecified(arr) {
		if (!Array.isArray(arr)) {
			throw new TypeError('Provide an array');
		}
		for (let i = 0; i <= arr.length; ++i) {
			window.localStorage.removeItem(arr[i]);
		}
	}

	/**
	 * Parse value to stringified JSON
	 * @param {*} value
	 * @return {string}
	 * @private
	 */
	_parseToString(value) {
		switch (typeof value) {
			case 'object' || 'array': {
				return JSON.stringify(value);
			}
			case 'map': {
				return JSON.stringify(Array.from(value.entries()));
			}
			default: {
				return JSON.stringify(value);
			}
		}
	}

	/**
	 * Parse stringified JSON back to original format
	 * @param {*} value
	 * @return {*}
	 * @private
	 */
	_parseToJSON(value) {
		return JSON.parse(value);
	}
}
